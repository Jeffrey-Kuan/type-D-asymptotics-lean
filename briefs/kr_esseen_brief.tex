\section*{Tier 1: lattice Esseen inversion bound}

For an integer-valued random variable $Y$ with characteristic function
$\varphi(t)=\E e^{\mathrm itY}$,
\[
  \Q(Y):=\sup_x\PP(Y=x)\ \le\ \frac1{2\pi}\int_{-\pi}^{\pi}|\varphi(t)|\,dt,
\]
directly from Fourier inversion on $\Z$:
$\PP(Y=x)=\frac1{2\pi}\int_{-\pi}^{\pi}e^{-\mathrm itx}\varphi(t)\,dt$
(orthogonality of characters; interchange of sum and integral is finite/
dominated).

\section*{Tier 2: Kolmogorov--Rogozin, product-over-gaps route}

For independent $Y_1,\dots,Y_n$ with $W:=\sum_j(1-\Q(Y_j))>0$, prove
$\Q(\sum Y_j)\le C/\sqrt W$ with universal $C$:
\begin{itemize}
\item[(a)] $|\varphi_S|=\prod_j|\varphi_j|$ and
$|\varphi_j|\le\exp\big(-\tfrac12(1-|\varphi_j|^2)\big)$ (from
$x\le e^{-(1-x^2)/2}$ on $[0,1]$).
\item[(b)] \underline{Symmetrization:} $1-|\varphi_j(t)|^2=\E(1-\cos tY^s_j)$
where $Y^s_j=Y_j-Y_j'$ ($Y_j'$ an independent copy); as a sum over the lattice,
\[
  1-|\varphi_j(t)|^2=\sum_{y\ne0}w_{j,y}\,(1-\cos ty),\qquad
  w_{j,y}:=\PP(Y^s_j=y)\ \ge0 .
\]
No probability space enlargement is needed: define
$w_{j,y}=\sum_x\PP(Y_j=x)\PP(Y_j=x+y)$ directly and prove the identity
algebraically from $|\varphi_j|^2=\varphi_j\overline{\varphi_j}$.
\item[(c)] \underline{Atom bound:} $\PP(Y^s_j=0)=\sum_x\PP(Y_j=x)^2\le\Q(Y_j)$,
hence $W_j:=\sum_{y\ne0}w_{j,y}\ge1-\Q(Y_j)$ and $\sum_jW_j\ge W$.
\item[(d)] \underline{Single-gap integral:} for each integer $y\ne0$ and $a>0$,
by $|y|$-periodicity of $u\mapsto1-\cos(yu)$ (substitute $v=yu$; the interval
$[-\pi,\pi]$ maps onto $|y|$ periods),
\[
  \int_{-\pi}^{\pi}e^{-a(1-\cos yt)}\,dt=\int_{-\pi}^{\pi}e^{-a(1-\cos v)}\,dv
  \ \le\ \frac{C_1}{\sqrt{1+a}},
\]
the latter from $1-\cos v\ge\tfrac2{\pi^2}v^2$ on $[-\pi,\pi]$ and the Gaussian
integral.
\item[(e)] \underline{Generalized H\"older:} truncate each gap sum to a finite
set carrying $W_j^L\uparrow W_j$; with weights $\lambda_{j,y}=w_{j,y}/W^L$,
$W^L:=\sum_{j,y\le L}w_{j,y}$,
\[
  \int_{-\pi}^{\pi}\prod_{j,y}e^{-\frac12w_{j,y}(1-\cos ty)}\,dt
  \ \le\ \prod_{j,y}\Big(\int_{-\pi}^{\pi}
  e^{-\frac12W^L(1-\cos ty)}\,dt\Big)^{w_{j,y}/W^L}
  \ \le\ \frac{C_1}{\sqrt{1+W^L/2}},
\]
by finite H\"older (\texttt{inner\_le\_nnorm}\dots use Mathlib's
finite-product H\"older / \texttt{integral\_prod\_norm\_pow\_le} or prove the
finite case by induction from Cauchy--Schwarz--type interpolation; the
exponents sum to $1$) and (d). Let $L\to\infty$, then combine with Tier 1 and
(c): $\Q(S)\le C/\sqrt{1+W}\le C/\sqrt W$.
\end{itemize}
Mind $W$ possibly $0$ for some $j$ (those factors contribute exponent $0$ ---
drop them) and the hypothesis $0<W$ of the target statement.
